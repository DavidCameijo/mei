class Solution:
    def kthSmallestPath(self, destination: List[int], k: int) -> str:
        from math import comb
        
        v = destination[0]
        h = destination[1]
        res = []
        
        while h > 0 or v > 0:
            if h == 0:
                res.append('V')
                v -= 1
            elif v == 0:
                res.append('H')
                h -= 1
            else:
                cnt = comb(h - 1 + v, h - 1)
                if k <= cnt:
                    res.append('H')
                    h -= 1
                else:
                    res.append('V')
                    k -= cnt
                    v -= 1
        
        return "".join(res)
