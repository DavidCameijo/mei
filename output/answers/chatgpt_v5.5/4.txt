from typing import List

class Solution:
    def findMissingAndRepeatedValues(self, grid: List[List[int]]) -> List[int]:
        n = len(grid)
        total = n * n
        seen = [0] * (total + 1)
        repeat = -1
        
        for row in grid:
            for x in row:
                seen[x] += 1
                if seen[x] == 2:
                    repeat = x
        
        missing = -1
        for i in range(1, total + 1):
            if seen[i] == 0:
                missing = i
                break
        
        return [repeat, missing]
