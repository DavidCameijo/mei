class Solution:
    def findTheLongestBalancedSubstring(self, s: str) -> int:
        n = len(s)
        best = 0

        for i in range(n):
            zeros = 0
            ones = 0
            phase = 0

            for j in range(i, n):
                if s[j] == '0':
                    if phase == 1:
                        break
                    zeros += 1
                else:
                    phase = 1
                    ones += 1

                if zeros > 0 and ones > 0 and zeros == ones:
                    best = max(best, 2 * zeros)

        return best
