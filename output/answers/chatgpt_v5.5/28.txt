class Solution:
    def splitArraySameAverage(self, nums):
        from math import gcd
        n = len(nums)
        S = sum(nums)
        
        dp = [set() for _ in range(n + 1)]
        dp[0].add(0)
        
        for num in nums:
            for k in range(n - 1, -1, -1):
                if dp[k]:
                    for s in list(dp[k]):
                        dp[k + 1].add(s + num)
        
        for k in range(1, n):
            if (S * k) % n == 0:
                target = (S * k) // n
                if target in dp[k]:
                    return True
        return False
