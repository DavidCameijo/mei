from typing import List
import heapq
import math

class Solution:
    def minimumTime(self, power: List[int]) -> int:
        n = len(power)
        full = (1 << n) - 1
        
        def needed_days(g, p):
            lo, hi = 0, 200000
            while lo < hi:
                mid = (lo + hi) // 2
                val = mid * g + mid * (mid - 1) // 2
                if val >= p:
                    hi = mid
                else:
                    lo = mid + 1
            return lo + 1
        
        INF = 10**18
        dist = [[INF] * (n + 2) for _ in range(1 << n)]
        
        start = (0, 1)
        dist[0][1] = 0
        pq = [(0, 0, 1)]
        
        while pq:
            d, mask, g = heapq.heappop(pq)
            if d != dist[mask][g]:
                continue
            if mask == full:
                return d
            
            for i in range(n):
                if mask & (1 << i):
                    continue
                
                t = needed_days(g, power[i])
                nd = d + t
                nm = mask | (1 << i)
                ng = g + t
                
                if nd < dist[nm][ng]:
                    dist[nm][ng] = nd
                    heapq.heappush(pq, (nd, nm, ng))
        
        return 0
