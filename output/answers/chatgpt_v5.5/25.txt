class Solution:
    def longestDupSubstring(self, s: str) -> str:
        n = len(s)
        nums = [ord(c) - 97 for c in s]
        
        def search(L):
            if L == 0:
                return 0
            base = 26
            mod1 = 1000000007
            mod2 = 1000000009
            
            h1 = 0
            h2 = 0
            for i in range(L):
                h1 = (h1 * base + nums[i]) % mod1
                h2 = (h2 * base + nums[i]) % mod2
            
            seen = {(h1, h2): [0]}
            
            baseL1 = pow(base, L, mod1)
            baseL2 = pow(base, L, mod2)
            
            for i in range(1, n - L + 1):
                h1 = (h1 * base - nums[i - 1] * baseL1 + nums[i + L - 1]) % mod1
                h2 = (h2 * base - nums[i - 1] * baseL2 + nums[i + L - 1]) % mod2
                
                key = (h1, h2)
                if key in seen:
                    for j in seen[key]:
                        if s[j:j+L] == s[i:i+L]:
                            return i
                    seen[key].append(i)
                else:
                    seen[key] = [i]
            return -1
        
        left, right = 1, n - 1
        start = 0
        best_len = 0
        
        while left <= right:
            mid = (left + right) // 2
            pos = search(mid)
            if pos != -1:
                start = pos
                best_len = mid
                left = mid + 1
            else:
                right = mid - 1
        
        return s[start:start + best_len]
