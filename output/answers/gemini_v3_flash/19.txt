class Solution:
    def maxPotholes(self, road: str, budget: int) -> int:
        groups = []
        count = 0
        for char in road:
            if char == 'x':
                count += 1
            else:
                if count > 0:
                    groups.append(count)
                count = 0
        if count > 0:
            groups.append(count)
        
        groups.sort(reverse=True)
        
        total_fixed = 0
        for g in groups:
            if budget <= 1:
                break
            
            cost_for_all = g + 1
            if budget >= cost_for_all:
                total_fixed += g
                budget -= cost_for_all
            else:
                can_fix = budget - 1
                total_fixed += can_fix
                budget = 0
                
        return total_fixed
