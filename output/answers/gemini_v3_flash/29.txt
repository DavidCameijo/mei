class Solution:
    def minimumTime(self, power: List[int]) -> int:
        n = len(power)
        dp = [float('inf')] * (1 << n)
        dp[0] = 0
        
        for mask in range(1 << n):
            if dp[mask] == float('inf'):
                continue
            
            # gain is the number of monsters defeated + 1
            gain = bin(mask).count('1') + 1
            
            for i in range(n):
                if not (mask & (1 << i)):
                    # Days to reach power[i] mana with current gain
                    days_needed = (power[i] + gain - 1) // gain
                    new_mask = mask | (1 << i)
                    if dp[mask] + days_needed < dp[new_mask]:
                        dp[new_mask] = dp[mask] + days_needed
                        
        return dp[(1 << n) - 1]
