class Solution:
    def champagneTower(self, poured: int, query_row: int, query_glass: int) -> float:
        res = [poured] + [0] * query_row
        for row in range(query_row):
            next_res = [0] * (query_row + 1)
            for glass in range(row + 1):
                excess = (res[glass] - 1.0) / 2.0
                if excess > 0:
                    next_res[glass] += excess
                    next_res[glass + 1] += excess
            res = next_res
        
        return min(1.0, res[query_glass])
