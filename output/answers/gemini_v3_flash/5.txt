# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def gameResult(self, head: Optional[ListNode]) -> str:
        even_points = 0
        odd_points = 0
        curr = head
        
        while curr:
            even_val = curr.val
            odd_val = curr.next.val
            
            if even_val > odd_val:
                even_points += 1
            elif odd_val > even_val:
                odd_points += 1
            
            curr = curr.next.next
            
        if even_points > odd_points:
            return "Even"
        elif odd_points > even_points:
            return "Odd"
        else:
            return "Tie"
