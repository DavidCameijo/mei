class Solution:
    def getMaxLen(self, nums: List[int]) -> int:
        pos = 0
        neg = 0
        max_len = 0
        
        for n in nums:
            if n > 0:
                pos += 1
                neg = neg + 1 if neg > 0 else 0
            elif n < 0:
                new_pos = neg + 1 if neg > 0 else 0
                new_neg = pos + 1
                pos, neg = new_pos, new_neg
            else:
                pos = 0
                neg = 0
            
            if pos > max_len:
                max_len = pos
                
        return max_len
