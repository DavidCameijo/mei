from typing import List
import heapq

class Solution:
    def findAnswer(self, n: int, edges: List[List[int]]) -> List[bool]:
        graph = [[] for _ in range(n)]
        for i, (a, b, w) in enumerate(edges):
            graph[a].append((b, w, i))
            graph[b].append((a, w, i))
        
        def dijkstra(src):
            dist = [float('inf')] * n
            dist[src] = 0
            pq = [(0, src)]
            while pq:
                d, u = heapq.heappop(pq)
                if d > dist[u]:
                    continue
                for v, w, _ in graph[u]:
                    if dist[u] + w < dist[v]:
                        dist[v] = dist[u] + w
                        heapq.heappush(pq, (dist[v], v))
            return dist
        
        dist0 = dijkstra(0)
        distn = dijkstra(n - 1)
        shortest = dist0[n - 1]
        
        answer = []
        for a, b, w in edges:
            on_path = (dist0[a] + w + distn[b] == shortest or
                       dist0[b] + w + distn[a] == shortest)
            answer.append(on_path)
        return answer
