class Solution:
    def maxPotholes(self, road: str, budget: int) -> int:
        groups = []
        count = 0
        for c in road + '.':
            if c == 'x':
                count += 1
            else:
                if count > 0:
                    groups.append(count)
                    count = 0
        
        groups.sort(reverse=True)
        
        total = 0
        for g in groups:
            cost = g + 1
            if budget >= cost:
                budget -= cost
                total += g
            else:
                can_fix = budget - 1
                if can_fix > 0:
                    total += can_fix
                break
        
        return total
