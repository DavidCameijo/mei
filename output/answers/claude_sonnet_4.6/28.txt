from typing import List

class Solution:
    def splitArraySameAverage(self, nums: List[int]) -> bool:
        n = len(nums)
        total = sum(nums)
        
        # For a subset of size k with sum s: s/k == total/n
        # => s*n == total*k
        # Check if any valid (k, s) exists
        
        half = n // 2
        left = nums[:n//2]
        right = nums[n//2:]
        
        def get_subsets(arr):
            subsets = [set() for _ in range(len(arr) + 1)]
            subsets[0].add(0)
            for num in arr:
                for k in range(len(arr), 0, -1):
                    for s in subsets[k-1]:
                        subsets[k].add(s + num)
            return subsets
        
        left_subsets = get_subsets(left)
        right_subsets = get_subsets(right)
        
        for k in range(1, n):
            if (total * k) % n != 0:
                continue
            target = (total * k) // n
            lk = min(k, len(left))
            for lsize in range(max(0, k - len(right)), lk + 1):
                rsize = k - lsize
                if rsize > len(right):
                    continue
                for ls in left_subsets[lsize]:
                    if (target - ls) in right_subsets[rsize]:
                        return True
        
        return False
