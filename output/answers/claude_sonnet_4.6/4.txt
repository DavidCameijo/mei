from typing import List
from collections import Counter

class Solution:
    def findMissingAndRepeatedValues(self, grid: List[List[int]]) -> List[int]:
        n = len(grid)
        count = Counter()
        for row in grid:
            for val in row:
                count[val] += 1
        a = b = 0
        for x in range(1, n*n+1):
            if count[x] == 2:
                a = x
            elif count[x] == 0:
                b = x
        return [a, b]
